Discutons de votre Projet

Projet : Gestion RH (M2L)

Le Besoin Métier

J'ai réalisé la refonte d'une application de gestion RH pour des ligues sportives. L'enjeu principal était de passer d'un outil rudimentaire à une solution multi-utilisateurs entièrement sécurisée.

Modélisation & Analyse

Pour structurer ce projet, j'ai commencé par une phase d'analyse avec un arbre heuristique pour lister les fonctionnalités. J'ai ensuite conçu le MCD et utilisé la modélisation UML pour garantir une base de données cohérente et une architecture logique robuste avant de débuter tout développement.

Sécurité & Contrôle d'accès

Le point clé de mon travail a été la mise en place d'un contrôle d'accès strict par rôle. Chaque utilisateur ne peut accéder qu'aux données correspondant à son niveau d'habilitation (Employé, Administrateur de ligue, Super-administrateur), ce qui garantit la confidentialité totale des informations sensibles de la ligue.

Projet : Skystorm (Laravel)

Le Concept

Mon second projet, Skystorm, est une application web sociale. L'objectif était de concevoir une plateforme interactive permettant aux utilisateurs de publier des messages, de suivre d'autres profils et d'interagir via un système de 'likes'.

Architecture & Base de données

Pour ce projet, j'ai utilisé le framework PHP Laravel 12 en respectant l'architecture MVC (Modèle-Vue-Contrôleur). J'ai débuté par la conception du MCD pour structurer une base de données MySQL capable de gérer les relations complexes entre les utilisateurs, les abonnements et les publications.

Points Clés de Réalisation

Code Source : Bot Anti-Nuke

1. Initialisation

Importation des dépendances, configuration des Intents Discord et vérification stricte des variables d'environnement (.env) pour empêcher un lancement compromis.

2. Alertes (Webhooks)

Utilisation d'un Webhook pour envoyer des alertes formatées (Embeds) dans un salon sécurisé sans dépendre de l'API classique des messages.

3. Analyse des Logs

La fonction checkAuditLog récupère le dernier "Audit Log" de Discord pour identifier formellement l'utilisateur à l'origine d'une action destructrice récente.

4. Défense Active

Cœur de la sécurité : la fonction nukeDefense retire instantanément tous les rôles modifiables à l'utilisateur malveillant pour bloquer sa capacité de nuisance (whitelist applicable).

5. Événements

Le bot surveille en continu le serveur. Si un salon est supprimé ou un membre expulsé massivement, il intercepte l'événement et déclenche la défense.

bot.js
require('dotenv').config();
const { 
    Client, 
    GatewayIntentBits, 
    AuditLogEvent, 
    WebhookClient, 
    EmbedBuilder 
} = require('discord.js');

const TOKEN = process.env.DISCORD_TOKEN;
const OWNER_ID = process.env.OWNER_ID;
const WEBHOOK_URL = process.env.WEBHOOK_URL;

// Vérification des variables
if (!TOKEN || !OWNER_ID || !WEBHOOK_URL) {
    console.error("ERREUR CRITIQUE : Variables manquantes dans le fichier .env");
    console.error("Vérifie : DISCORD_TOKEN, OWNER_ID, WEBHOOK_URL");
    process.exit(1);
}

const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMembers,
        GatewayIntentBits.GuildModeration,
        GatewayIntentBits.GuildWebhooks,
        GatewayIntentBits.GuildPresences
    ]
});

const webhook = new WebhookClient({ url: WEBHOOK_URL });

async function sendAlert(titre, description, color = 0xFF0000) {
    const embed = new EmbedBuilder()
        .setTitle(`SECURITE : ${titre}`)
        .setDescription(description)
        .setColor(color)
        .setTimestamp()
        .setFooter({ text: 'Anti-Nuke System' });

    webhook.send({
        username: 'Sentinel Bot',
        avatarURL: client.user?.displayAvatarURL(),
        embeds: [embed]
    }).catch(console.error);
}

async function checkAuditLog(guild, typeAudit) {
    await new Promise(r => setTimeout(r, 2000)); 

    const fetchedLogs = await guild.fetchAuditLogs({
        limit: 1,
        type: typeAudit,
    });
    const log = fetchedLogs.entries.first();

    if (!log) return null;
    
    const now = Date.now();
    if (now - log.createdTimestamp > 15000) return null;

    return log.executor;
}

async function nukeDefense(guild, executorId, actionType) {
    if (executorId === OWNER_ID || executorId === client.user.id) return;

    try {
        const member = await guild.members.fetch(executorId);
        if (!member) return;

        const rolesRetirables = member.roles.cache.filter(role => 
            role.name !== '@everyone' && 
            role.editable &&              
            !role.managed                
        );

        if (rolesRetirables.size > 0) {
            await member.roles.remove(rolesRetirables);
            
            console.log(`PUNITION : ${rolesRetirables.size} roles retires a ${member.user.tag}`);

            await sendAlert(
                'MENACE NEUTRALISEE',
                `**Coupable :** <@${executorId}> (${member.user.tag})\n` +
                `**Action :** ${actionType}\n` +
                `**Sanction :** ${rolesRetirables.size} roles retires.`
            );
        } else {
            console.log(`Alerte : ${member.user.tag} a fait une action mais je n'ai aucun role a lui retirer.`);
            sendAlert(
                'ACTION SUSPECTE (SANS SANCTION)',
                `ATTENTION : <@${executorId}> a fait : **${actionType}**.\nJe n'ai pas pu retirer de roles (Grade trop haut ou roles proteges).`,
                0xFFA500
            );
        }

    } catch (error) {
        console.error(`Erreur punition:`, error);
        sendAlert('ERREUR TECHNIQUE', `Impossible de punir <@${executorId}> : ${error.message}`, 0x555555);
    }
}

client.once('ready', () => {
    console.log(`${client.user.tag} est en ligne !`);
    sendAlert('SYSTEME EN LIGNE', 'Le bot est actif sur le Serveur Local.', 0x00FF00);
});

client.on('channelDelete', async (channel) => {
    const executor = await checkAuditLog(channel.guild, AuditLogEvent.ChannelDelete);
    if (executor) nukeDefense(channel.guild, executor.id, `Suppression salon #${channel.name}`);
});

client.on('channelCreate', async (channel) => {
    const executor = await checkAuditLog(channel.guild, AuditLogEvent.ChannelCreate);
    if (executor) nukeDefense(channel.guild, executor.id, `Creation salon #${channel.name}`);
});

client.on('channelUpdate', async (oldChannel, newChannel) => {
    if (oldChannel.name !== newChannel.name) {
        const executor = await checkAuditLog(newChannel.guild, AuditLogEvent.ChannelUpdate);
        if (executor) nukeDefense(newChannel.guild, executor.id, `Renommage salon #${oldChannel.name} -> #${newChannel.name}`);
    }
});

client.on('roleDelete', async (role) => {
    const executor = await checkAuditLog(role.guild, AuditLogEvent.RoleDelete);
    if (executor) nukeDefense(role.guild, executor.id, `Suppression role ${role.name}`);
});

client.on('roleCreate', async (role) => {
    const executor = await checkAuditLog(role.guild, AuditLogEvent.RoleCreate);
    if (executor) nukeDefense(role.guild, executor.id, `Creation role ${role.name}`);
});

client.on('guildBanAdd', async (ban) => {
    const executor = await checkAuditLog(ban.guild, AuditLogEvent.MemberBanAdd);
    if (executor) nukeDefense(ban.guild, executor.id, `Bannissement d'un membre`);
});

client.on('guildMemberRemove', async (member) => {
    const executor = await checkAuditLog(member.guild, AuditLogEvent.MemberKick);
    if (executor && executor.target?.id === member.id) {
        nukeDefense(member.guild, executor.id, `Expulsion (Kick) de ${member.user.tag}`);
    }
});

client.on('guildMemberAdd', async (member) => {
    if (member.user.bot) {
        const executor = await checkAuditLog(member.guild, AuditLogEvent.BotAdd);
        if (executor) nukeDefense(member.guild, executor.id, `Ajout bot non autorise ${member.user.tag}`);
    }
});

client.on('webhooksUpdate', async (channel) => {
    const executor = await checkAuditLog(channel.guild, AuditLogEvent.WebhookCreate);
    if (executor) nukeDefense(channel.guild, executor.id, `Creation Webhook`);
});

client.on('guildUpdate', async (oldG, newG) => {
    const executor = await checkAuditLog(newG, AuditLogEvent.GuildUpdate);
    if (executor) nukeDefense(newG, executor.id, `Modification Serveur`);
});

client.login(TOKEN);
